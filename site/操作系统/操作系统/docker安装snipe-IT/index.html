    version: "3"
services:
  snipeit:
    image: snipeit #镜像版本
    ports:
    - "80:80"
    container_name: snipeit
    volumes:
    - ./logs:/var/www/html/storage/logs
    - ./apache2.log:/var/log/apache2/
    env_file:
    - snipeit.env
    networks:
    - snipeit-backend
    restart: unless-stopped
    ulimits:
      nofile:
        soft: 1048576
        hard: 1048576
      nproc: 104857
volumes:
  db: {}

networks:
  snipeit-backend: {}


docker-compose up -d    #启动服务
